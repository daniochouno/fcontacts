static Future<List<FContact>> list({query: String}) async {
  Iterable filtered =
      await _channel.invokeMethod("list", <String, dynamic>{'query': query});
  var mapped = filtered.map((item) => FContact.fromMap(item));
  return mapped.toList();
}
